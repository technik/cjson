42  inline Json::Json(const Json& _x)
43  : mType (_x.mType)
44  {
45  switch(mType) {
46  case DataType::boolean:
47  case DataType::integer:
48  case DataType::real:
49  mNumber = _x.mNumber;
50  break;
51  case DataType::text:
52  mText = _x.mText;
53  break;
54  case DataType::array:
55  mArray.reserve(_x.mArray.size());
56  for(auto element : _x.mArray)
57  mArray.push_back(new Json(*element));
58  break;
59  case DataType::object:
60  for(auto element : _x.mObject)
61  mObject.insert(std::make_pair(element.first, new Json(*element.second)));
62  break;
63  default: // Do nothing for null
64  break;
65  }
66  }
